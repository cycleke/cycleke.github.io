public class App implements Runnable {
  public static int count = 0;
  private static boolean lock = false;
  private final int id;

  public App(int id) {
    this.id = id;
  }

  @Override
  public void run() {
    System.out.println(id + " " + ++count);
    try {
      Thread.sleep(id);
    } catch (Exception e) {
      e.printStackTrace();
    } finally {
      System.out.println(id + " " + ++count);
    }
  }

  public static void main(String[] args) {
    int n = 5;
    App[] apps = new App[n];
    for (int i = 0; i < n; ++i)
      apps[i] = new App(i);
    for (int i = n - 1; i >= 0; --i) {
      new Thread(apps[i]).start();
    }
    System.out.println(App.count);
  }
}

  public static int count = 0;
  private static boolean lock = false;
  private final int id;

  public App(int id) {
    this.id = id;
  }

  private int increaseCount() {
    while (lock)
      continue;
    lock = true;
    int res = ++count;
    lock = false;

  public void run() {
